BFS & DFS

BFS (Breadth-First Search): ‚Ä¢ Uses: Queue (FIFO) ‚Ä¢ Explores: Level by level ‚Ä¢ Finds: Shortest path (unweighted) ‚Ä¢ Time: O(V + E) DFS (Depth-First Search): ‚Ä¢ Uses: Stack/Recursion ‚Ä¢ Explores: As deep as possible first ‚Ä¢ Finds: Paths, cycles, components ‚Ä¢ Time: O(V + E)
// BFS
function bfs(graph, start) {
    const visited = new Set([start]);
    const queue = [start];
    while (queue.length) {
        const node = queue.shift();
        for (const neighbor of graph[node]) {
            if (!visited.has(neighbor)) {
                visited.add(neighbor);
                queue.push(neighbor);
            }
        }
    }
}

// DFS
function dfs(graph, node, visited = new Set()) {
    visited.add(node);
    for (const neighbor of graph[node]) {
        if (!visited.has(neighbor)) {
            dfs(graph, neighbor, visited);
        }
    }
}

Shortest Paths

DIJKSTRA (non-negative weights): ‚Ä¢ Single source shortest path ‚Ä¢ Uses priority queue (min-heap) ‚Ä¢ Time: O((V + E) log V) with heap BELLMAN-FORD (allows negative weights): ‚Ä¢ Single source, can detect negative cycles ‚Ä¢ Relaxes all edges V-1 times ‚Ä¢ Time: O(V √ó E) FLOYD-WARSHALL (all pairs): ‚Ä¢ Shortest paths between all pairs ‚Ä¢ Dynamic programming approach ‚Ä¢ Time: O(V¬≥)
// Dijkstra (simplified with array)
function dijkstra(graph, start) {
    const dist = {};
    for (let v in graph) dist[v] = Infinity;
    dist[start] = 0;
    const pq = [[0, start]]; // [distance, node]
    
    while (pq.length) {
        pq.sort((a,b) => a[0] - b[0]);
        const [d, u] = pq.shift();
        if (d > dist[u]) continue;
        for (const [v, w] of graph[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push([dist[v], v]);
            }
        }
    }
    return dist;
}

Minimum Spanning Tree

MST: Subset of edges connecting all vertices with minimum total weight.
KRUSKAL'S ALGORITHM: 1. Sort edges by weight 2. Add edges smallest first 3. Skip if would create cycle (union-find) Time: O(E log E) PRIM'S ALGORITHM: 1. Start from any vertex 2. Greedily add cheapest edge to new vertex 3. Uses priority queue Time: O((V + E) log V)
// Kruskal with Union-Find
function kruskal(vertices, edges) {
    edges.sort((a, b) => a.weight - b.weight);
    const parent = {};
    const find = x => parent[x] === x ? x : (parent[x] = find(parent[x]));
    const union = (a, b) => { parent[find(a)] = find(b); };
    
    vertices.forEach(v => parent[v] = v);
    const mst = [];
    
    for (const {u, v, weight} of edges) {
        if (find(u) !== find(v)) {
            union(u, v);
            mst.push({u, v, weight});
        }
    }
    return mst;
}
